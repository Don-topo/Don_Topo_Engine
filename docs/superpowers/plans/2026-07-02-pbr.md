# PBR Rendering Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Phong shading with Cook-Torrance GGX PBR on both static and skinned pipelines, adding optional glTF ORM texture (binding 4) and ACES tone mapping inline in the fragment shader.

**Architecture:** Push constants extended to 80 bytes (mat4 + metallic + roughness + pad). Both pipelines share `m_descriptorSetLayout` and `m_pipelineLayout`. A new `pbr.frag` shader replaces `triangle.frag` in both pipelines. ORM texture loaded from asset if present; 1×1 white fallback otherwise. No new render passes.

**Tech Stack:** Vulkan 1.2, GLSL 450, GLM, Assimp, stb_image, glslc.

## Global Constraints

- C++20, MSVC, Ninja build via CMakePresets
- Push constant size must stay ≤ 128 bytes (Vulkan minimum guarantee)
- Both `m_pipeline` (static) and `m_skinnedGfxPipeline` share `m_descriptorSetLayout` and `m_pipelineLayout` — updating them affects both
- No new render passes, no new framebuffers, no IBL
- Compile shaders with `glslc shaders/pbr.frag -o shaders/pbr.frag.spv` then copy to `build-ninja/sandbox/shaders/`
- Build: open a VS 2022 x64 Native Tools terminal, `cd c:\Users\ruben\Documents\Don_Topo_Engine`, then `cmake --preset windows-ninja && cmake --build build-ninja`

```cpp
    void Renderer::createSolidColorImage(const uint8_t rgba[4], VkImage& img, VkDeviceMemory& mem)
    {
        VkBuffer sb; VkDeviceMemory sm;
        createBuffer(4,
            VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
            VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
            sb, sm);
        void* mapped;
        vkMapMemory(m_device, sm, 0, 4, 0, &mapped);
        memcpy(mapped, rgba, 4);
        vkUnmapMemory(m_device, sm);
        createImage(1, 1, VK_FORMAT_R8G8B8A8_UNORM, VK_IMAGE_TILING_OPTIMAL,
            VK_IMAGE_USAGE_TRANSFER_DST_BIT | VK_IMAGE_USAGE_SAMPLED_BIT,
            VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT, img, mem);
        transitionImageLayout(img, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
        copyBufferToImage(sb, img, 1, 1);
        transitionImageLayout(img, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
        vkDestroyBuffer(m_device, sb, nullptr);
        vkFreeMemory(m_device, sm, nullptr);
    }
```

### Task 6: pbr.frag shader — write, compile, deploy

**Files:**
- Create: `shaders/pbr.frag`

**Interfaces:**
- Consumes: vertex outputs from `triangle.vert` (locations 0-6), UBO at binding 0, texSampler at binding 1, normalMap at binding 2, shadowMap at binding 3, metallicRoughnessTex at binding 4; push constants `push.metallic`, `push.roughness`
- Produces: `outColor` = PBR lit + ACES tone mapped + gamma corrected

- [ ] **Step 1: Create shaders/pbr.frag**

Create `shaders/pbr.frag` with the following content:

```glsl
#version 450

layout(location = 0) in vec3 fragColor;
layout(location = 1) in vec2 fragUV;
layout(location = 2) in vec3 fragNormal;
layout(location = 3) in vec3 fragWorldPos;
layout(location = 4) in vec3 fragTangent;
layout(location = 5) in vec3 fragBitangent;
layout(location = 6) in vec4 fragLightSpacePos;

layout(location = 0) out vec4 outColor;

#define MAX_LIGHTS 4
struct Light { vec4 position; vec4 color; };

layout(set = 0, binding = 0) uniform UBO {
    mat4  view;
    mat4  proj;
    mat4  lightSpaceMatrix;
    Light lights[MAX_LIGHTS];
    vec4  viewPos;
    int   numLights;
} ubo;

layout(set = 0, binding = 1) uniform sampler2D texSampler;
layout(set = 0, binding = 2) uniform sampler2D normalMap;
layout(set = 0, binding = 3) uniform sampler2DShadow shadowMap;
layout(set = 0, binding = 4) uniform sampler2D metallicRoughnessTex;

layout(push_constant) uniform PushData {
    mat4  transform;
    float metallic;
    float roughness;
    vec2  _pad;
} push;

const float PI = 3.14159265359;

float computeShadow(vec4 lightSpacePos)
{
    vec3 proj = lightSpacePos.xyz / lightSpacePos.w;
    proj.xy   = proj.xy * 0.5 + 0.5;
    if (proj.z > 1.0 || proj.z < 0.0) return 1.0;
    vec2 texelSize = 1.0 / vec2(2048.0);
    float shadow = 0.0;
    for (int x = -1; x <= 1; x++)
        for (int y = -1; y <= 1; y++)
            shadow += texture(shadowMap, vec3(proj.xy + vec2(x, y) * texelSize, proj.z));
    return shadow / 9.0;
}

vec3 aces(vec3 x)
{
    return clamp((x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14), 0.0, 1.0);
}

void main()
{
    // TBN + normal map
    mat3 TBN  = mat3(fragTangent, fragBitangent, fragNormal);
    vec3 N    = normalize(TBN * (texture(normalMap, fragUV).rgb * 2.0 - 1.0));
    vec3 V    = normalize(ubo.viewPos.xyz - fragWorldPos);

    // Albedo (linearize from sRGB)
    vec3 albedo = pow(texture(texSampler, fragUV).rgb, vec3(2.2));

    // ORM: R=AO, G=roughness, B=metallic — multiply by push scalars
    vec3  orm      = texture(metallicRoughnessTex, fragUV).rgb;
    float ao       = orm.r;
    float rough    = clamp(orm.g * push.roughness, 0.04, 1.0);
    float metal    = clamp(orm.b * push.metallic,  0.0,  1.0);

    vec3 F0 = mix(vec3(0.04), albedo, metal);

    float shadow = computeShadow(fragLightSpacePos);
    vec3  Lo     = vec3(0.0);

    for (int i = 0; i < ubo.numLights; i++)
    {
        vec3  L        = normalize(ubo.lights[i].position.xyz - fragWorldPos);
        vec3  H        = normalize(V + L);
        float dist     = length(ubo.lights[i].position.xyz - fragWorldPos);
        float atten    = 1.0 / (dist * dist);
        vec3  radiance = ubo.lights[i].color.rgb * ubo.lights[i].color.a * atten;

        float NdotL = max(dot(N, L), 0.0);
        float NdotV = max(dot(N, V), 0.0);
        float NdotH = max(dot(N, H), 0.0);
        float HdotV = max(dot(H, V), 0.0);

        // D — GGX
        float a  = rough * rough;
        float a2 = a * a;
        float d  = (NdotH * NdotH) * (a2 - 1.0) + 1.0;
        float D  = a2 / (PI * d * d);

        // G — Smith-Schlick
        float r = rough + 1.0;
        float k = (r * r) / 8.0;
        float G = (NdotV / (NdotV * (1.0 - k) + k)) * (NdotL / (NdotL * (1.0 - k) + k));

        // F — Schlick
        vec3 F  = F0 + (1.0 - F0) * pow(clamp(1.0 - HdotV, 0.0, 1.0), 5.0);

        vec3 kD = (1.0 - F) * (1.0 - metal);

        float s = (i == 0) ? shadow : 1.0;

        Lo += s * (kD * albedo / PI + D * G * F / (4.0 * NdotV * NdotL + 0.0001))
              * radiance * NdotL;
    }

    vec3 ambient = vec3(0.03) * albedo * ao;
    vec3 color   = ambient + Lo;

    outColor = vec4(pow(aces(color), vec3(1.0 / 2.2)), 1.0);
}
```
